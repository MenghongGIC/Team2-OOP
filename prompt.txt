# See what’s going on (best git command ever)
git status -sb

# See colored, compact log with graph
git log --oneline --graph --decorate --all

# See last 10 commits nicely
git log --oneline -10

# See what you’ve changed but not staged
git diff

# See what you’ve staged
git diff --staged

# Add everything (careful!) or just specific files
git add .
git add path/to/file.js

# Commit with a good message (opens editor)
git commit

# Commit with message directly
git commit -m "fix: correct typo in login button"

# Amend last commit (fix message or add forgotten files)
git commit --amend --no-edit   # just add files
git commit --amend             # change message too

# Create and switch to new branch
git checkout -b feature/cool-thing

# Or (newer syntax)
git switch -c feature/cool-thing

# Push branch + set upstream
git push -u origin feature/cool-thing

# Delete local branch
git branch -d old-branch

# Delete remote branch
git push origin --delete old-branch

# Unstage a file
git restore --staged file.js

# Discard changes in a file (dangerous!)
git restore file.js

# Undo last commit but keep changes staged
git reset --soft HEAD~1

# Undo last commit and unstage everything
git reset HEAD~1

# Totally remove last commit (history rewrite)
git reset --hard HEAD~1

# "Oh no I deleted something!" → recover it
git reflog                     # find the hash
git checkout -b rescue-branch <hash>

# Remove untracked files/folders
git clean -fd                  # dry-run first: git clean -fdn

# Fixup previous commit (add new changes to last commit)
git commit --fixup HEAD
git rebase -i --autosquash HEAD~2

# Create commit that undoes a previous one
git revert <commit-hash>

# Stash with message
git stash push -m "work in progress - fixing auth"

# List and apply stashes
git stash list
git stash apply stash@{1}

# Nuke everything and go back to clean main
git fetch origin
git reset --hard origin/main
git clean -fd
